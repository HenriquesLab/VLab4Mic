350    def generate_imaging(
351        self,
352        modality=None,
353        channel="ch0",
354        nframes=1,
355        nbeads=0,
356        save=False,
357        noise=False,
358        exp_time=1.0,
359        **kwargs,
360    ):
361        """
362        Master funciton that generates image sequences depending on the modality
363        This function is responsible for
364            Preparing the coordinates according to ROI (DONE!)
365            Creating the Photons per frame matrix
366            Take focus plane, PSF and photon budgets
367            Feed these data into the Convolution method (or analytical)
368            Adds noise
369            Profit!
370        """
371        if modality is None:
372            modality = list(self.modalities.keys())[0]
373        # get fluorophores to simulate
374        print(f"Simulating imaging from modality: {modality} in channel {channel}")
375
376        fluonames = list(self.modalities[modality]["filters"][channel])
377        # prepare a dictionary that conaints the emitters per channel defined
378        output_per_fluoname = dict()
379        writing_notes = self.identifier + "_" + str(modality) + "_" + str(channel) + "_"
380        for fluo in fluonames:  # a channel could capture multiple fluorophores
381            # print(fluo)
382            writing_notes_fluo = writing_notes + str(fluo)
383            emitters = self.get_emitters_in_ROI(fluo)
384            n_emitters = emitters.shape[0]
385            if n_emitters < 1:
386                # if no emitter is in range, an image with noise should be generated
387                emitters = np.array(
388                    [
389                        [0, 0, 0],
390                    ]
391                )
392                photons_frames = np.repeat(0, nframes).reshape((1, nframes))
393                emission_notes = "None"
394                print(f"photons should be zero: {photons_frames}")
395                field_data, psf_data = self._homogenise_scales4convolution_modality(
396                    modality, emitters, photons_frames
397                )
398
399            else:
400                # adjust the absolute XY positions to the ROI dimentions
401                emitters[:, 0] = emitters[:, 0] - self.roi_params["ranges"][0][0]
402                emitters[:, 1] = emitters[:, 1] - self.roi_params["ranges"][1][0]
403                photons_frames, emission_notes = self.calculate_photons_per_frame(
404                    modality, fluo, n_emitters, nframes, exp_time
405                )
406                field_data, psf_data = self._homogenise_scales4convolution_modality(
407                    modality, emitters, photons_frames
408                )
409            # write emitter positions after being placed in the FOV
410            gt_notes = writing_notes_fluo + "_usedForImaging"
411            emitters_to_export = field_data["field_coordinates"]
412            if save:
413                self.write_ground_truth_positions(
414                    emitters_to_export, "x [nm],y [nm],z [nm]", gt_notes
415                )
416            convolution_type = self.modalities[modality]["psf"]["convolution_type"]
417            psf_size = psf_data["psf_array"].shape
418            print(f"size of psf is: {psf_size}")
419            simparams = dict(
420                field_data=field_data,
421                psf_data=psf_data,
422                nbeads=nbeads,
423                photons_per_bead=self.fluorophore_params[fluo]["blinking"][
424                    "photons_per_blink"
425                ],
426                psf_projection_depth=self.modalities[modality]["psf"]["depth"],
427            )
428            if convolution_type == "direct":
429                pass
430            else:
431                images, beads = self.images_by_convolutions(
432                    convolution_type, **simparams
433                )
434            writing_notes_fluo = writing_notes_fluo + emission_notes
435            # print(f"max and min only photons: {np.max(images)}, {np.min(images)}")
436            # Beta implementation to save images before detection
437            if save:
438                self._save_timeseries_with_beads(images, beads, writing_notes_fluo)
439
440            # wrap up the images from a single fluorophore in the channel
441            output_per_fluoname[fluo] = dict(images=images, beads=beads)
442        timeseries, beadstack = self._add_fluorophore_signals(output_per_fluoname)
443        # # # Up to here only the photon information on arrival
444        if noise:
445            print("Adding noise")
446            timeseries = self._crop_negative(timeseries)
447            # print(np.max(timeseries), np.min(timeseries))
448            timeseries = self.add_detector_noise(modality, timeseries)
449            if beadstack is not None:
450                beadstack = self.add_detector_noise(modality, beadstack)
451            else:
452                beadstack = None
453        if save:
454            # gt_positions = self.generate_ground_truth_positions(groundtruth_emitters)
455            timeseries = self._crop_negative(timeseries)
456            beadstack = self._crop_negative(beadstack)
457            writing_notes_fluo = writing_notes_fluo + "_withNoise_"
458
459            self._save_timeseries_with_beads(timeseries, beadstack, writing_notes_fluo)
460        return timeseries, beadstack

484    def _add_fluorophore_signals(self, output_per_fluoname: dict):
485        """
486        Dictionary has the fluorophores as keys, each fluorophore
487        will have the timeseries and can also have the stack of beads
488        if exists
489        """
490        fluorophores = list(output_per_fluoname.keys())
491        shp_ims = output_per_fluoname[fluorophores[0]]["images"].shape
492        timeseries = np.zeros((shp_ims))
493        for fl in fluorophores:
494            both_images = np.array([timeseries, output_per_fluoname[fl]["images"]])
495            timeseries = np.sum(both_images, axis=0)
496        if output_per_fluoname[fluorophores[0]]["beads"] is not None:
497            shp_bds = output_per_fluoname[fluorophores[0]]["beads"].shape
498            beadstack = np.zeros((shp_bds))
499            for fl in fluorophores:
500                both_beads = np.array([beadstack, output_per_fluoname[fl]["beads"]])
501                beadstack = np.sum(both_beads, axis=0)
502        else:
503            beadstack = None
504        return timeseries, beadstack

506    def add_detector_noise(self, modality, stack):
507        """
508        INput is an image stack and a dictionary with the
509        Types of noise to add, and the parameters needed for each one
510        The Noise model also includes an order of noise addition
511        """
512        # consider all the parameters of the detection system and corrupt
513        # the photon data
514        noise_params = self.modalities[modality]["detector"]["noise_model"]
515        noise_order = self.modalities[modality]["detector"]["noise_order"]
516        for noise_type in noise_order:
517            # print(noise_type)
518            stack = add_image_noise(noise_type, stack, **noise_params[noise_type])
519            # includes the processes from collecting the photons, multiplying
520            # and all possible source of noise
521            # plus the ADU conversion factor
522        # Efectively is the digitalisation part and sets the saturation level
523        stack = self._adjust_to_pixel_depth(modality, stack)
524        return stack

648        """

671    def _homogenise_scales4convolution_modality(
672        self, modality, emitters_in_channel, photons_frames, **kwargs
673    ):
674        """
675        Compiles all data necesary to carry on the generation of frames
676        considering the input coordinates
677        the calclated photons per frame
678        and the PSF of the input modality
679        and expresses all dimensions in the dimensions of the PSF
680
681        For example, if the image generator scale is in micrometers
682        all the coordinates will be set to nanometers, which is the expected
683        scale for the pixelsize of the PSF
684
685        Ideally, the resulting dictionary is passed to a custom convolution
686        function
687
688        """
689        # check if scaling is needed
690        psf_scale = self.modalities[modality]["psf"]["scale"]
691        # print(f"psf scale: psf_scale")
692        scaling = math.ceil(self.get_roi_params("scale") / psf_scale)
693        # print(f"scaling for converting field data for convolution: {scaling}")
694        # define field values
695        scaling_used = scaling
696        field_size = self.modalities[modality]["detector"][
697            "image_size"
698        ]  # in pixels xyz
699        field_pixelsizeXY = (
700            self.modalities[modality]["detector"]["pixelsize"] * scaling
701        )  # detector pixelsize
702        field_coordinates = emitters_in_channel * scaling
703        field_zfocus = self.get_roi_params("focus_plane") * scaling
704        # ALL PARAMETERS REGARDING FIELD ARE NOW IN PSF SCALE UNITS
705        field_scale = psf_scale
706        field_data = dict(
707            field_scale=field_scale,
708            field_size=field_size,
709            field_pixelsizeXY=field_pixelsizeXY,
710            field_coordinates=field_coordinates,
711            field_zfocus=field_zfocus,
712            photons_frames=photons_frames,
713        )
714        # psf parameters
715        psf_scale = psf_scale
716        psf_array = self.modalities[modality]["psf"]["psf_stack"]
717        psf_zstep = self.modalities[modality]["psf"]["voxelsize"][
718            2
719        ]  # dimensions are assumed to be xyz
720        psf_pixelsizeXY = self.modalities[modality]["psf"]["voxelsize"][
721            0
722        ]  # equal pixelsize laterally
723        psf_focus_slice = int(self.modalities[modality]["psf"]["focus_plane"])
724        psf_data = dict(
725            psf_scale=psf_scale,
726            psf_array=psf_array,
727            psf_zstep=psf_zstep,
728            psf_pixelsizeXY=psf_pixelsizeXY,
729            psf_focus_slice=psf_focus_slice,
730        )
731        # print(f"Matching PSF and coordinates scales")
732        return field_data, psf_data
