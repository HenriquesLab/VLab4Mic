 113        Set the ideal label locations and information for label targeting.

 201        self,
 202        targets: dict,
 203        scale: float = 1e-10,
 204        axis=None,
 205        label_name="NA",
 206        labelling_efficiency: float = 1.0,
 207        minimal_distance: float = 0.0,
 208        secondary=False,
 209        info="NA",
 210        **kwargs,
 211    ):
 212        """
 213        Add a label to the instance.
 214
 215        Parameters
 216        ----------
 217        targets : dict
 218            Dictionary with fluorophore names as keys and emitter coordinates as values.
 219        scale : float, optional
 220            Scale of the label. Default is 1e-9.
 221        axis : dict, optional
 222            Axis definition for the label.
 223        label_name : str, optional
 224            Name of the label.
 225        labelling_efficiency : float, optional
 226            Labelling efficiency. Default is 1.0.
 227        minimal_distance : float, optional
 228            Minimal distance between emitters. Default is 0.0.
 229        secondary : bool, optional
 230            If True, label is secondary. Default is False.
 231        info : str, optional
 232            Additional information.
 233        **kwargs
 234            Additional keyword arguments.
 235        """
 236        # Set default axis if None is provided or if axis has None values
 237        if axis is None:
 238            axis = dict(pivot=[0, 0, 0], direction=[0, 0, 1])
 239        else:
 240            # Ensure axis has valid values, not None
 241            if axis.get("pivot") is None:
 242                axis["pivot"] = [0, 0, 0]
 243            if axis.get("direction") is None:
 244                axis["direction"] = [0, 0, 1]
 245
 246        # should check if the name is different
 247        label_name = label_name
 248        self.labelnames.append(label_name)
 249        for (
 250            keys,
 251            values,
 252        ) in targets.items():  # assumption: theres only one key and value
 253            fluorophore = keys
 254            emitters = values
 255        if emitters is None:
 256            label_type = "direct"
 257        else:
 258            label_type = "indirect"
 259            # add minimal distance to radial hindrance
 260            probe_max_dist = np.max(pdist(emitters))
 261            new_hindrance = self.radial_hindance + 2 * probe_max_dist
 262            if new_hindrance > self.radial_hindance:
 263                self.radial_hindance = new_hindrance
 264        colour = set_colorplot(self.plotting_params)
 265        # print(f"colour assigned to label: {colour}")
 266        plt_params = dict(
 267            plotsize=20, plotalpha=1, plotmarker="o", plotcolour=colour
 268        )
 269        self.plotting_params[label_name] = plt_params
 270        label_params = dict(
 271            emitters=emitters,
 272            scale=scale,
 273            axis=axis,
 274            labelling_efficiency=labelling_efficiency,
 275            minimal_distance=minimal_distance,
 276            label_type=label_type,
 277            fluorophore=fluorophore,
 278        )
 279        for key, val in kwargs.items():
 280            label_params[key] = val
 281        if secondary:
 282            self.secondary[label_name] = label_params
 283        else:
 284            self.labels[label_name] = label_params
 285            self.status["labels"] = True

 694    def transform_reorient_axis(self, neworientation: np.array, reset_orientation=False):
 695        """
 696        Reorient the instance to a new orientation.
 697
 698        Parameters
 699        ----------
 700        neworientation : numpy.ndarray
 701            New orientation vector.
 702        """
 703        if reset_orientation:
 704            self.reset_axis_orientation()
 705        else:
 706            if np.linalg.norm(neworientation) == 0:
 707                print(
 708                    f"Norm for vector {neworientation} is 0. No reorientation done"
 709                )
 710            else:
 711                thet = np.arccos(
 712                    np.dot(self.axis["direction"], neworientation)
 713                    / (
 714                        np.linalg.norm(self.axis["direction"])
 715                        * np.linalg.norm(neworientation)
 716                    )
 717                )
 718                if np.absolute(thet) == 1:
 719                    print(
 720                        f"input vector {neworientation} "
 721                        f'has same direction as {self.axis["direction"]}. '
 722                        f"No reorientation done"
 723                    )
 724                else:
 725                    nori = copy.copy(neworientation)
 726                    # this function should take the new orientation and match the
 727                    for trgt in self.labelnames:
 728                        if self.get_emitter_by_target(trgt) is not None:
 729                            reoriented = rotate_pts_by_vector(
 730                                self.get_emitter_by_target(trgt),
 731                                self.axis["direction"],
 732                                nori,
 733                                self.get_ref_point(),
 734                            )
 735                            self.emitters[trgt] = reoriented
 736                            self.axis["direction"] = nori

 695    def transform_reorient_axis(self, neworientation: np.array, reset_orientation=False):
 696        """
 697        Reorient the instance to a new orientation.
 698
 699        Parameters
 700        ----------
 701        neworientation : numpy.ndarray
 702            New orientation vector.
 703        """
 704        if reset_orientation:
 705            self.reset_axis_orientation()
 706        else:
 707            if np.linalg.norm(neworientation) == 0:
 708                print(
 709                    f"Norm for vector {neworientation} is 0. No reorientation done"
 710                )
 711            else:
 712                thet = np.arccos(
 713                    np.dot(self.axis["direction"], neworientation)
 714                    / (
 715                        np.linalg.norm(self.axis["direction"])
 716                        * np.linalg.norm(neworientation)
 717                    )
 718                )
 719                if np.absolute(thet) == 1:
 720                    print(
 721                        f"input vector {neworientation} "
 722                        f'has same direction as {self.axis["direction"]}. '
 723                        f"No reorientation done"
 724                    )
 725                else:
 726                    nori = copy.copy(neworientation)
 727                    # this function should take the new orientation and match the
 728                    for trgt in self.labelnames:
 729                        if self.get_emitter_by_target(trgt) is not None:
 730                            reoriented = rotate_pts_by_vector(
 731                                self.get_emitter_by_target(trgt),
 732                                self.axis["direction"],
 733                                nori,
 734                                self.get_ref_point(),
 735                            )
 736                            self.emitters[trgt] = reoriented
 737                            self.axis["direction"] = nori
