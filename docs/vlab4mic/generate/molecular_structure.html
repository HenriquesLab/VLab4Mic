 178        """
 179        Generate a random substring of a given size from a string.
 180
 181        Parameters
 182        ----------
 183        string : str
 184            Input string.
 185        size : int, optional

 188        Returns

 190        str

 192        """

 195            end = start + size

 253    def generate_assemmbly_operations(self):
 254        """
 255        Parse the rotation/translation operations needed to construct a molecular assembly from an asymmetric unit.
 256        """
 257        if self.CIFdictionary is None:  # check if already created
 258            self.generate_MMCIF_dictionary()
 259        if "AlphaFoldDB" in self.CIFdictionary["_database_2.database_id"]:
 260            self.assymetric_defined = False
 261        else:
 262            # get to know how many transformations there are
 263            transformation_ids = self.CIFdictionary["_pdbx_struct_oper_list.id"]
 264            # each element of the list is pair of matrix-vector needed to transform data
 265            # iterate over each of the transformations
 266            assembly_transformations = []
 267            for tr in range(len(transformation_ids)):
 268                row1 = [
 269                    self.CIFdictionary["_pdbx_struct_oper_list.matrix[1][1]"][tr],
 270                    self.CIFdictionary["_pdbx_struct_oper_list.matrix[1][2]"][tr],
 271                    self.CIFdictionary["_pdbx_struct_oper_list.matrix[1][3]"][tr],
 272                ]
 273
 274                row2 = [
 275                    self.CIFdictionary["_pdbx_struct_oper_list.matrix[2][1]"][tr],
 276                    self.CIFdictionary["_pdbx_struct_oper_list.matrix[2][2]"][tr],
 277                    self.CIFdictionary["_pdbx_struct_oper_list.matrix[2][3]"][tr],
 278                ]
 279
 280                row3 = [
 281                    self.CIFdictionary["_pdbx_struct_oper_list.matrix[3][1]"][tr],
 282                    self.CIFdictionary["_pdbx_struct_oper_list.matrix[3][2]"][tr],
 283                    self.CIFdictionary["_pdbx_struct_oper_list.matrix[3][3]"][tr],
 284                ]
 285                rotation_matrix = np.array([row1, row2, row3], dtype="float")
 286
 287                vector = np.array(
 288                    [
 289                        self.CIFdictionary["_pdbx_struct_oper_list.vector[1]"][tr],
 290                        self.CIFdictionary["_pdbx_struct_oper_list.vector[2]"][tr],
 291                        self.CIFdictionary["_pdbx_struct_oper_list.vector[3]"][tr],
 292                    ],
 293                    dtype="float",
 294                )
 295                assembly_transformations.append([rotation_matrix, vector])
 296            struct_oper_dictionary = dict(zip(transformation_ids, assembly_transformations))
 297            if len(transformation_ids) > 1:
 298                self.assymetric_defined = True
 299                # print(
 300                #    "This model is defined with more than one symmetric transformation.
 301                # Will consider the assembly as assymetric defined"
 302                # )
 303            else:
 304                # when no assembly unit exist, there is only 1 transform expected
 305                self.assymetric_defined = False
 306                # print("This model is defined with only one symmetric transformation")
 307            self.assembly_operations = struct_oper_dictionary

 388    def _central_axis_from_rotations(self):
 389        """
 390        Calculate the central axis orientation from the assembly operations.
 391
 392        Returns
 393        -------
 394        numpy.ndarray
 395            Central axis vector.
 396        """
 397        average = np.identity(3)
 398        keys = list(self.assembly_operations.keys())
 399        for i in range(1, len(self.assembly_operations)):
 400            average = np.matmul(average, self.assembly_operations[keys[i]][0])
 401        eigenvalues, eigenvectors = np.linalg.eig(average)
 402        central_axis = eigenvectors[
 403            :, np.argmax(np.isclose(eigenvalues, 1.0))
 404        ]  # central axis is the eigenvector that corresponds to the eigenvalue = 1
 405        if central_axis.dtype == "complex128":
 406            central_axis = np.array([0, 0, 1])
 407        return central_axis

389    def _central_axis_from_rotations(self):
390        """
391        Calculate the central axis orientation from the assembly operations.
392
393        Returns
394        -------
395        numpy.ndarray
396            Central axis vector.
397        """
398        average = np.identity(3)
399        keys = list(self.assembly_operations.keys())
400        for i in range(1, len(self.assembly_operations)):
401            average = np.matmul(average, self.assembly_operations[keys[i]][0])
402        eigenvalues, eigenvectors = np.linalg.eig(average)
403        central_axis = eigenvectors[
404            :, np.argmax(np.isclose(eigenvalues, 1.0))
405        ]  # central axis is the eigenvector that corresponds to the eigenvalue = 1
406        if central_axis.dtype == "complex128":
407            central_axis = np.array([0, 0, 1])
408        return central_axis
