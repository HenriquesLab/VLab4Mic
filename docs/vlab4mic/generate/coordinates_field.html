 414    def _random_pos_minimal_dist(self, n):
 415        # convert minimal distance in relative units
 416        selected_positions = []
 417        selected_relative = []
 418        minimal_distance = self.get_molecule_param("minimal_distance")
 419        dimension_sizes = self.get_field_param("dimension_sizes")
 420        maxabs_posx = dimension_sizes[0]
 421        maxabs_posy = dimension_sizes[1]
 422        maxabs_posz = dimension_sizes[2]
 423        # print(f"Max values: {maxabs_posx},{maxabs_posy}, {maxabs_posz}")
 424        # sample the first point
 425        x = np.random.uniform(0, maxabs_posx, size=1)
 426        y = np.random.uniform(0, maxabs_posy, size=1)
 427        z = np.random.uniform(0, maxabs_posz, size=1)
 428        v0 = np.array([x, y, z]).reshape(-1)
 429        selected_positions.append(v0)
 430        flag = 0
 431        while len(selected_positions) < n and flag < 10000:
 432            flag = flag + 1
 433            x = np.random.uniform(0, maxabs_posx, size=1)
 434            y = np.random.uniform(0, maxabs_posy, size=1)
 435            z = np.random.uniform(0, maxabs_posz, size=1)
 436            new = np.array([x, y, z]).reshape(-1)
 437            is_available = 1
 438            for pos in range(len(selected_positions)):
 439                # verify is the next point is within the minimum distance
 440                # print(np.linalg.norm(new - selected_positions[pos]))
 441                if np.linalg.norm(new - selected_positions[pos]) < minimal_distance:
 442                    is_available = 0
 443                    # break
 444            if is_available:
 445                selected_positions.append(new)
 446        for abs_pos in selected_positions:
 447            rel_pos = abs_pos
 448            rel_pos[0] = abs_pos[0] / maxabs_posx
 449            rel_pos[1] = abs_pos[1] / maxabs_posy
 450            rel_pos[2] = abs_pos[2] / maxabs_posz
 451            selected_relative.append(rel_pos)
 452        # print(f"end flag : {flag}")
 453        self.molecules_params["relative_positions"] = selected_relative
 454        # self.set_molecule_param("absolute_positions", selected_positions)
 455        # self.absolute_pos = selected_positions

 516        if self.molecules_params["minimal_distance"] is None:

 528            print(f"Updating Number of molecules: {reps}")

415    def _random_pos_minimal_dist(self, n):
416        # convert minimal distance in relative units
417        selected_positions = []
418        selected_relative = []
419        minimal_distance = self.get_molecule_param("minimal_distance")
420        dimension_sizes = self.get_field_param("dimension_sizes")
421        maxabs_posx = dimension_sizes[0]
422        maxabs_posy = dimension_sizes[1]
423        maxabs_posz = dimension_sizes[2]
424        # print(f"Max values: {maxabs_posx},{maxabs_posy}, {maxabs_posz}")
425        # sample the first point
426        x = np.random.uniform(0, maxabs_posx, size=1)
427        y = np.random.uniform(0, maxabs_posy, size=1)
428        z = np.random.uniform(0, maxabs_posz, size=1)
429        v0 = np.array([x, y, z]).reshape(-1)
430        selected_positions.append(v0)
431        flag = 0
432        while len(selected_positions) < n and flag < 10000:
433            flag = flag + 1
434            x = np.random.uniform(0, maxabs_posx, size=1)
435            y = np.random.uniform(0, maxabs_posy, size=1)
436            z = np.random.uniform(0, maxabs_posz, size=1)
437            new = np.array([x, y, z]).reshape(-1)
438            is_available = 1
439            for pos in range(len(selected_positions)):
440                # verify is the next point is within the minimum distance
441                # print(np.linalg.norm(new - selected_positions[pos]))
442                if np.linalg.norm(new - selected_positions[pos]) < minimal_distance:
443                    is_available = 0
444                    # break
445            if is_available:
446                selected_positions.append(new)
447        for abs_pos in selected_positions:
448            rel_pos = abs_pos
449            rel_pos[0] = abs_pos[0] / maxabs_posx
450            rel_pos[1] = abs_pos[1] / maxabs_posy
451            rel_pos[2] = abs_pos[2] / maxabs_posz
452            selected_relative.append(rel_pos)
453        # print(f"end flag : {flag}")
454        self.molecules_params["relative_positions"] = selected_relative
455        # self.set_molecule_param("absolute_positions", selected_positions)
456        # self.absolute_pos = selected_positions
