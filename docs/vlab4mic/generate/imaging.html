 576        self,
 577        modality=None,

 579        nframes=1,

 581        save=False,
 582        noise=False,
 583        exp_time=0.001,
 584        masks=False,
 585        **kwargs,
 586    ):
 587        """
 588        Master function that generates image sequences depending on the modality
 589        This function is responsible for
 590            Preparing the coordinates according to ROI (DONE!)
 591            Creating the Photons per frame matrix
 592            Take focus plane, PSF and photon budgets
 593            Feed these data into the Convolution method (or analytical)
 594            Adds noise
 595            Profit!
 596        """
 597        if modality is None:
 598            modality = list(self.modalities.keys())[0]
 599        # get fluorophores to simulate
 600        print(f"Simulating imaging from modality: {modality} in channel {channels}")
 601        output_per_channel = dict()
 602        output_per_channel_noiseless = dict()
 603        beads_per_channel = dict()
 604        beads_per_channel_noiseless = dict()
 605        for ch in channels:
 606            if ch not in self.modalities[modality]["filters"].keys():
 607                raise ValueError(
 608                    f"Channel {ch} not found in modality {modality} filters"
 609                )
 610            else:
 611                fluonames = list(self.modalities[modality]["filters"][ch])
 612                # prepare a dictionary that contains the emitters per channel defined
 613                output_per_fluoname = dict()
 614                writing_notes = self.identifier + "_" + str(modality) + "_" + str(ch) + "_"
 615                for fluo in fluonames:  # a channel could capture multiple fluorophores
 616                    # print(fluo)
 617                    writing_notes_fluo = writing_notes + str(fluo)
 618                    if masks:
 619                        emitters = self.get_emitters_in_ROI(fluo, masks=masks)
 620                    else:
 621                        emitters = self.get_emitters_in_ROI(fluo)
 622                    n_emitters = emitters.shape[0]
 623                    if n_emitters < 1:
 624                        no_emitters = True
 625                        # if no emitter is in range, an image with noise should be generated
 626                        emitters = np.array(
 627                            [
 628                                [0, 0, 0],
 629                            ]
 630                        )
 631                        photons_frames = np.repeat(0, nframes).reshape((1, nframes))
 632                        emission_notes = "None"
 633                        print(f"photons should be zero: {photons_frames}")
 634                        field_data, psf_data = self._homogenise_scales4convolution_modality(
 635                            modality, emitters, photons_frames
 636                        )
 637
 638                    else:
 639                        no_emitters = False
 640                        # adjust the absolute XY positions to the ROI dimentions
 641                        emitters[:, 0] = emitters[:, 0] - self.roi_params["ranges"][0][0]
 642                        emitters[:, 1] = emitters[:, 1] - self.roi_params["ranges"][1][0]
 643                        photons_frames, emission_notes = self.calculate_photons_per_frame(
 644                            modality, fluo, n_emitters, nframes, exp_time
 645                        )
 646                        field_data, psf_data = self._homogenise_scales4convolution_modality(
 647                            modality, emitters, photons_frames
 648                        )
 649                    # write emitter positions after being placed in the FOV
 650                    gt_notes = writing_notes_fluo + "_usedForImaging"
 651                    emitters_to_export = field_data["field_coordinates"]
 652                    if save:
 653                        self.write_ground_truth_positions(
 654                            emitters_to_export, "x [nm],y [nm],z [nm]", gt_notes, no_emitters

1071

1081            "image_size"
1082        ]  # in pixels xyz

1084            self.modalities[modality]["detector"]["pixelsize"] * scaling
1085        )  # detector pixelsize

1087        field_zfocus = self.get_roi_params("focus_plane") * scaling

1102            2
