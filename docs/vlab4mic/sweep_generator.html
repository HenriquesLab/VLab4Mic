 463    def set_parameter_values(self, param_group, param_name, values=None):
 464        """
 465        Specify list or range of values to use per parameter.
 466
 467        Parameters
 468        ----------
 469        :param param_group: str
 470            Group of parameter, e.g., 'probe', 'virtual_sample'.
 471        :param param_name: str
 472            Name of parameter to use for sweep.
 473        :param values: None, tuple, or list, optional
 474            Parameter values to use. If tuple, interpreted as (start, stop, num) for linspace.
 475            If list, used directly. If None, defaults are generated.
 476
 477        Returns
 478        -------
 479        None
 480        """
 481        if param_group in list(self.params_by_group.keys()):
 482            if values is None:
 483                # create defaults or ignore this parameter for it to not be used
 484                pass
 485            elif type(values) == list:
 486                # set list directly as the params to use
 487                self.params_by_group[param_group][param_name] = values
 488            elif type(values) == tuple:
 489                # 3 values are expected: min, max, steps
 490                # generate a linspace
 491                if param_group in self.parameter_settings.keys():
 492                    if (
 493                        param_name
 494                        in self.parameter_settings[param_group].keys()
 495                    ):
 496                        if (
 497                            self.parameter_settings[param_group][param_name][
 498                                "wtype"
 499                            ]
 500                            == "int_slider"
 501                        ):
 502                            step = np.ceil((values[1] - values[0]) / values[2])
 503                            self.params_by_group[param_group][param_name] = (
 504                                np.arange(
 505                                    start=values[0],
 506                                    stop=values[1],
 507                                    step=step,
 508                                    dtype=int,
 509                                )
 510                            )
 511                        else:
 512                            param_iterables = np.linspace(
 513                                values[0], values[1], values[2]
 514                            )
 515                            self.params_by_group[param_group][
 516                                param_name
 517                            ] = param_iterables
 518            self.parameters_with_set_values.append(param_name)
 519        else:
 520            print(f"{param_group} is not a valid parameter group")
