 206def sweep_modalities_updatemod(
 207    experiment: ExperimentParametrisation = None,

 213):
 214    """
 215    Simulate image acquisitions for all virtual samples and modalities.
 216
 217    Parameters
 218    ----------
 219    experiment : ExperimentParametrisation, optional
 220        Experiment object to use. If None, a new one is created.
 221    vsample_outputs : dict, optional

 235        The experiment object used.

 239        Dictionary of parameter combinations for each modality.
 240    pixelsizes : dict
 241        Dictionary of pixel sizes for each modality.
 242    """
 243    default_mod = "Confocal"
 244    default_aqc = dict(nframes=2, exp_time=0.005)

 247    mod_outputs_masks = dict()

 250        experiment = ExperimentParametrisation()
 251    if vsample_outputs is None:
 252        vsample_outputs = [
 253            default_vsample,
 254        ]
 255        # needs default sample. can be a minimal field with a single emitter
 256    experiment.clear_modalities()

 283        list(vsampl_pars.keys()),
 284        position=0,
 285        leave=True,
 286        desc="Unique parameter combination",

 305                            acq_pars,

 315
 316                            mod_comb = (

 318                                + "_"
 319                                + str(mod_n)
 320                                + "_"
 321                                + str(mod_pars_number)
 322                                + "_"
 323                                + str(mod_acq_number)

 328                                experiment.imager.modalities[modality_name][
 329                                    "detector"
 330                                ]["pixelsize"]
 331                                * 1000
 332                            )

 334                            mod_params_copy["pixelsize"] = pxsize

 346                            mask = vsample_mask_per_mod > 0
 347                            mod_outputs_masks[mod_comb].append(
 348                                mask
 349                            )
 350                            mod_parameters = None
 351                    mod_n += 1
 352    return experiment, mod_outputs, mod_params, pixelsizes, mod_outputs_masks

 497def analyse_image_sweep(
 498    img_outputs, img_params, reference, analysis_case_params=None
 499):
 500    """
 501    Analyse a sweep of images against a reference image.
 502
 503    Parameters
 504    ----------
 505    img_outputs : dict
 506        Dictionary of simulated image outputs.
 507    img_params : dict
 508        Dictionary of image parameters.
 509    reference : numpy.ndarray
 510        Reference image.
 511    analysis_case_params : dict, optional
 512        Additional parameters for analysis.
 513
 514    Returns
 515    -------
 516    measurement_vectors : list
 517        List of measurement results for each image.
 518    inputs : dict
 519        Dictionary of input images and used references.
 520    """
 521    measurement_vectors = []
 522    # ref_pixelsize = analysis_case_params["ref_pixelsize"]
 523    inputs = dict()
 524    for params_id in img_params.keys():
 525        inputs[params_id] = dict()
 526        rep_number = 0
 527        mod_name = img_params[params_id][5]  # 5th item corresponds to Modality
 528        for img_r in img_outputs[params_id]:
 529            im1 = img_r[0]
 530            im_ref = reference[0]
 531            rep_measurement, ref_used, qry_used_, masks_used = metrics.img_compare(
 532                im_ref, im1, **analysis_case_params[mod_name]
 533            )
 534            measurement_vectors.append(
 535                [params_id, rep_number, rep_measurement]
 536            )
 537            inputs[params_id][rep_number] = [qry_used, im1]
 538            rep_number += 1
 539    return measurement_vectors, inputs

 542def analyse_sweep_single_reference(
 543    img_outputs,
 544    img_outputs_masks,
 545    img_params,
 546    reference_image,
 547    reference_image_mask,
 548    reference_params,
 549    zoom_in=0,
 550    metrics_list: list = [
 551        "ssim",
 552    ],
 553    **kwargs,
 554):
 555    """
 556    Analyse a sweep of images against a single reference image using specified metrics.
 557
 558    Parameters
 559    ----------
 560    img_outputs : dict
 561        Dictionary of simulated image outputs.
 562    img_params : dict
 563        Dictionary of image parameters.
 564    reference_image : numpy.ndarray
 565        Reference image.
 566    reference_params : dict
 567        Parameters for the reference image.
 568    zoom_in : float, optional
 569        Zoom factor for analysis. Defaults to 0.
 570    metrics_list : list of str, optional
 571        List of metric names to compute. Defaults to ["ssim"].
 572    **kwargs
 573        Additional keyword arguments.
 574
 575    Returns
 576    -------
 577    measurement_vectors : list
 578        List of measurement results for each image.
 579    inputs : dict
 580        Dictionary of input images and used references.
 581    metrics_list : list
 582        List of metric names used.
 583    """
 584    measurement_vectors = []
 585    # ref_pixelsize = analysis_case_params["ref_pixelsize"]
 586    inputs = dict()
 587    for params_id in img_params.keys():
 588        inputs[params_id] = dict()
 589        rep_number = 0
 590        mod_name = img_params[params_id][5]  # 5th item corresponds to Modality
 591        modality_pixelsize = img_params[params_id][6]["pixelsize"]
 592        for img_r, img_mask in zip(img_outputs[params_id], img_outputs_masks[params_id]):
 593            im1 = img_r[0]
 594            im1_mask = img_mask
 595            #print(f"query: {im1.shape},{im1_mask.shape}")
 596            im_ref = reference_image
 597            rep_measurement, ref_used, qry_used, masks_used = metrics.img_compare(
 598                ref = im_ref,
 599                ref_mask=reference_image_mask,
 600                query=im1,
 601                query_mask=im1_mask,
 602                modality_pixelsize=modality_pixelsize,
 603                ref_pixelsize=reference_params["ref_pixelsize"],
 604                force_match=True,
 605                zoom_in=zoom_in,
 606                metric=metrics_list,
 607            )
 608            r_vector = list([params_id, rep_number]) + list([*rep_measurement])
 609            measurement_vectors.append(r_vector)
 610            # measurement_vectors = measurement_vectors + rep_measurement[0]
 611            inputs[params_id][rep_number] = [qry_used, im1, ref_used, masks_used]
 612            rep_number += 1
 613    return measurement_vectors, inputs, metrics_list
